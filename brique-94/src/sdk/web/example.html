Molam Form - Web SDK Example

1. Basic Integration

Simply include the SDK script and add the <molam-checkout> element:

<script src="molam-form.js"></script>

<molam-checkout
  publishable-key="pk_test_1234567890abcdef"
  amount="99.99"
  currency="USD"
  customer-email="customer@example.com"
  customer-name="John Doe"
  description="Premium Plan Subscription"
></molam-checkout>

3. Programmatic Usage

You can also use the SDK programmatically:

// Initialize SDK
const molam = new Molam('pk_test_1234567890abcdef');

// Create payment intent
const intent = await molam.createPaymentIntent({
  amount: 100.00,
  currency: 'USD',
  customer_email: 'customer@example.com',
  description: 'Order #12345'
});

// Tokenize card
const paymentMethod = await molam.tokenizePaymentMethod({
  number: '4242424242424242',
  expMonth: '12',
  expYear: '2025',
  cvc: '123',
  name: 'John Doe'
});

// Confirm payment
const result = await molam.confirmPaymentIntent(
  intent.intent_reference,
  paymentMethod.token
);

console.log('Payment successful:', result);

4. Event Handling

Listen to payment events:

const checkout = document.querySelector('molam-checkout');

checkout.addEventListener('payment-success', (event) => {
  console.log('Payment succeeded:', event.detail);
  // Redirect to success page, show confirmation, etc.
  window.location.href = '/thank-you?transaction=' + event.detail.intent_reference;
});

checkout.addEventListener('payment-error', (event) => {
  console.error('Payment failed:', event.detail);
  // Show error message, retry logic, etc.
  alert('Payment failed: ' + event.detail.error);
});
